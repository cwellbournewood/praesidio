01# Strict PII anonymisation
02id: pii-strict
03version: v2026.05.3
04fail: closed
05rules:
06 - match: finding.label == "pii.email"
07 then: tokenise(vault="acme-vault")
08 - match: finding.label.startsWith("secrets.")
09 then: block(reason="credential leakage")
10 - match: principal.region == "EU"
11 then: route(model="azure/gpt-4o-eu")
12default: allow
13signature:
14 cosign: cosign.sigstore.dev/by/data-protection@acme
